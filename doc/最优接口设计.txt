* 扩展属性schema
*/

 * 扩展属性，方便扩展
 */
private Map<String, String> extAttrs = new HashMap<>();

// 可编程对象
public class ProgrammableObject<T> extends Extensible implements Programmable {
/**
* 对象编码
*/
private String code;

/**
 * 父对象编码
 */
private String fatherCode;

/**
 * 默认值
 */
private T defaultValue;
 
/**
 * 排序号
 */
private Integer sortNo;
}

// 动态属性
public DynamicAttribute extends ProgrammableObject<String> {
private String code;
private String name = ""; // 名称
// 动态属性可选值列表
private List<DynamicAttributerOption> options = new ArrayList<>();
// 动态属性类型
private DynamicAttributeType dynAttrType = DynamicAttributeType.Integer;
// 动态属性的值，拥有计算实例化的值，存在算法
private String value;
// 动态属性的Logic
private List<Logic> valueLogics;
}

// 动态属性可选值
public class DynamicAttributerOption extends ProgrammableObject<String> {

}

// 规格定义基类
public abstract class SpecificationDefinition {
    protected String id;
    protected String name;
    protected SpecType type;
    protected String code;
    protected Constraint constraints;
}

// 规格类型枚举
public enum SpecType {
    SIMPLE, COMPOSITE, RANGE, ENUM, INTERFACE
}

 

// 简单规格定义
public class SimpleSpecDef extends SpecificationDefinition {
    protected String unit;
    protected String precision;
    protected Boolean allowApproximate;
}

// 复合规格定义
public class CompositeSpecDef extends SpecificationDefinition {
    protected List<ComponentSpecDef> components;
}

// 组件规格定义
public class ComponentSpecDef extends SpecificationDefinition {
    protected String parentId; // 所属复合规格ID
}

// 范围规格定义
public class RangeSpecDef extends SpecificationDefinition {
    protected NumericValue min;
    protected NumericValue max;
    protected String unit;
}

// 接口规格定义
public class InterfaceSpecDef extends SpecificationDefinition {
    protected String portType;
    protected Integer count;
    protected String technology;
    protected List<InterfaceSpecDef> subInterfaces;
    protected List<String> powerTypes;
}

// 数值类
public class NumericValue {
    protected Double numericValue;
    protected String unit;
    protected Boolean isApproximate;
}

// 约束类
public class Constraint {
    protected Double min;
    protected Double max;
    protected List<String> allowedValues;
}

// 产品系列
public class ProductFamily {
    protected String code;
    protected String name;
    protected List<SpecificationDefinition> specificationDefs;
}

// 规格值基类
public abstract class SpecificationValue {
    protected String specId;
}

// 简单规格值
public class SimpleValue extends SpecificationValue {
    protected NumericValue numericValue;
    protected String stringValue;
}

// 复合规格值
public class CompositeValue extends SpecificationValue {
    protected Map<String, Object> components; // key为组件ID，value为对应的规格值
}

// 范围规格值
public class RangeValue extends SpecificationValue {
    protected NumericValue minValue;
    protected NumericValue maxValue;
    protected String unit;
}

// 接口规格值
public class InterfaceValue extends SpecificationValue {
    protected List<InterfacePort> ports;
}

// 接口端口
public class InterfacePort {
    protected String portType;
    protected Integer count;
    protected String technology;
    protected List<InterfacePort> subInterfaces;
    protected List<String> powerTypes;
}

public class ProductInstance {
    protected String id;
    protected String familyId;
    protected ProductFamily family;
    protected Map<SpecificationDefinition, SpecificationValue> specification;

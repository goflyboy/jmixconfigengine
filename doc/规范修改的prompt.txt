class ModulePacker {
	 
	
	//对算法进行打包
	String pack(Class cpClazz,String  outRootDir) {
		module = ModuleGenneratorByAnno.build(moduleAlgClazz);
		return pack(module, cpClazz, outRootDir);
	}

	String pack(Module module, Class cpClazz,String  outRootDir) {
		//根据packageName构建全路径名
		inputDir = CommHelper.getTestResource(cpClazz) + packageNamePath 
		
		
		根据module的code和id在outRootDir创建这个module的子文件夹（已经存在要删除）outDir  cp-{moduleCode}-{moduleId}，如：cp-m001-0001

		调用ModuleGenneratorByAnno.saveToFile(module,outDir + cp-{moduleCode}-{moduleId}.base.json),例如：cp-m001-0001.base.json
		
		isMultifile = isMultifile(cpClass)
		//如果isMultifile = true，则 inputDir的文件都需要打包，有ConstraintAlg类和测试类两个文件
		//如果isMultifile = false, 则inputDir的文件还包含其它工程， cpClazz既有ConstraintAlg类又有测试类 
		
		打运行的class包 packageClassJar 
		打源码包 packageSourceJar 
		return outDir;
	}	
		 
		 
	 
	boolean isMultifile(Class cpClassName) {
		如果cpClass继承了ConstraintAlgImpl则是多个文件，例如： com.jmix.scenario.hello.HelloConstraint (HelloConstraint extends ConstraintAlgImpl)
		则是多文件
		否则就是单文件， com.jmix.scenario.ruletest.ParaIntegerTest (ParaIntegerTest extends ModuleScenarioTestBase) 
	}

	boolean packClassJar(inputDir,module,isMultifile,outDir) { 
		如果isMultifile = true， 需要将ConstraintAlg类及测试类对应的class打成jar包就可以（如：HelloConstraint.class、HelloConstraintTest.class）
		如果isMultifile = false， 需要将ConstraintAlg类和其父类对应的class文件打成jar包就 （如：ParaIsHiddenTest.class、ParaIsHiddenTest$ParaIsHiddenConstraint.class）
		jar保存的文件格式：cp-{moduleCode}-{moduleId}.jar, 如：cp-m001-0001.jar 

	}


	boolean packSourceJar(inputDir,module,isMultifile,outDir) { 
		如果isMultifile = true， 需要将ConstraintAlg类及测试类就可以（如：HelloConstraint.java、HelloConstraintTest.java）
		如果isMultifile = false， 需要将测试类（含ConstraintAlg类） 文件打成jar包就 （如：ParaIsHiddenTest.java，内部包含了ParaIsHiddenConstraint）
		jar保存的文件格式：cp-{moduleCode}-{moduleId}-sources.jar, 如：cp-m001-0001-sources.jar
	}
	
}
